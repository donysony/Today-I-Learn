#### ㅊㅗㅣㅂㅣㄴㄱㅏㅂㅅ ㄱㅜㅎㅏㄱㅣ
```java
class Solution {
    public int solution(int[] array) {
        int[] arr = new int[1000];
        int maxValue = 0;
        int maxIdx = 0;
        for(int i=0; i<array.length;i++){
            int val = array[i];
            ++arr[val];

            if(maxValue<arr[val]){
                maxValue=arr[val];
                maxIdx = val;
            }else if(maxValue == arr[val]){
                maxIdx = -1;
            }


        }


        int answer = maxIdx;
        return answer;
    }
}
```
